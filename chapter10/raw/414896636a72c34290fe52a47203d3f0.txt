    var ratio = w/h;
    if (isNaN(ratio))
    {
        return h;
    }
    else
    {
        return winwidth / ratio;
    }

        if (liveStreamConfig && current_stream) {
            try {
               window.dataLayer.push({
                    'event': 'dataLayer-loaded',
                    'persistent-player-stream': liveStreamConfig[current_stream]['slug']
                });

                $('#stream-station').html(liveStreamConfig[current_stream][1]).attr('class', 'stream-' + liveStreamConfig[current_stream]['slug']);
                $('#launch-popout-player').data('width', liveStreamConfig[current_stream]['popup']['width'])
                        .data('height', liveStreamConfig[current_stream]['popup']['height'])
                        .data('url', liveStreamConfig[current_stream]['popup']['url']);

                $('#stream-controls').attr('class', 'control-' + liveStreamConfig[current_stream]['slug']);

                initializeLiveStreamPlayer();
            }
            catch(e){
                Rollbar.error("Livestream Player", e);
            }
        }
    }

    function toggleMoreStreams()
    {
        $('#persistent-streams-more .up').toggle();
        $('#persistent-streams-more .down').toggle();
    }

    // setup stream switcher
    var stream_switcher = '';
    $.each(liveStreamConfig, function (key, s) {
        stream_switcher += '<a class="stream-option stream-' + s.slug + '" data-streamid="' + s.id + '">' + s.titlePersistentPlayer + '</a>';
    });

    $('#persistent-streams-list').html(stream_switcher);

    /* ------------ */
    /* CLICK EVENTS */
    /* ------------ */
    $('#persistent-streams-more').click(function () {
        $('#persistent-streams-list').slideToggle();
        toggleMoreStreams();
    });

    $('#persistent_player_play').click(function () {
        if (liveStreamPlayer)
        {
            try {
                liveStreamPlayer.playlistItem(0);
                liveStreamPlayer.setVolume(100);
            }
            catch(e){
                Rollbar.error("Livestream Player", e);
            }
        }
    });

    $('#persistent_player_pause').click(function () {
        if (liveStreamPlayer) {
            try {
                liveStreamPlayer.pause();
            }
            catch(e){
                Rollbar.error("Livestream Player", e);
            }
        }
    });

    $('#persistent_player_play, #persistent_player_pause').click(function () {
        $('#persistent-player').toggleClass('playing');
    });

    $('.stream-option').click(function () {
        if (liveStreamPlayer) {
            try {
                $('#persistent-streams-list').slideToggle();
                liveStreamPlayer.stop();
                $('#persistent-player').removeClass('playing');
                current_stream = $(this).data('streamid');
                setupStream();
                toggleMoreStreams();
            }
            catch(e){
                Rollbar.error("Livestream Player", e);
            }
        }
    });

    $('#launch-popout-player').click(function() {
        if (liveStreamPlayer) {
             liveStreamPlayer.stop();
        }

        var w = $(this).data('width');
        var h = $(this).data('height');
        var url = $(this).data('url');
        $('#persistent-player').removeClass('playing');

        window.open(url, 'xspfplayer', 'width=' + w + 'px,height=' + h + ',status=no')
    });

    /* ------------ */
    /* SETUP        */
    /* ------------ */
    $(document).ready(function () {
        setupStream();
    });
</script>

			<p>Former South Korean President Lee Myung-bak has been arrested over a host of corruption allegations, ranging from bribery to embezzlement and tax evasion. Live footage <a href="https://twitter.com/jlinfan888/status/976837723398066177">on local media</a> showed Lee submitting to his arrest warrant at home late at night, getting into a black sedan flanked by two law enforcement&nbsp;officers.</p><p>&#8220;I feel that all is my fault,&#8221; Lee said in a <a href="https://www.facebook.com/leemyungbak/posts/1283148005163428">handwritten statement</a> posted to Facebook shortly after the warrant was issued, &#8220;and I feel a sense of&nbsp;guilt.&#8221;</p><p>That said, the 76-year-old conservative politician has denied the charges against him, with Lee rejecting the investigation as &#8220;political revenge&#8221; mounted by the current administration in office, headed by liberal President Moon&nbsp;Jae-in.</p><p>Lee, who led the country from 2008 to 2013, is just the latest high-profile South Korean politician to find himself embroiled in legal woes. His conservative successor, Park Geun-hye, saw her own presidency <a href="https://www.npr.org/2017/03/11/519807707/what-s-next-for-south-korea-after-it-ousted-its-president">unravel last year</a> under the weight of a massive graft scandal of her own. He killed himself while the probe was ongoing in May of that&nbsp;year.&#8221;</p><p></blockquote></p><p>Lee, for his part, stands accused of accepting millions of dollars in bribes from a variety of individuals and organizations  including, <a href="https://apnews.com/d357a7b6151c4aeeb6c0e6e3b8ca4d4f/Ex-S.-Korean-President-Lee-arrested-over-corruption-scandal">according to The Associated Press</a>, his own intelligence agency. Yonhap notes he is also alleged to have embezzled tens of millions more from a private&nbsp;company.</p><p>&#8220;Other charges include abuse of power, breach of trust, illegally stashing presidential documents and a violation of election law,&#8221; Yonhap&nbsp;explains.</p><p>All told, Lee faces up to 45 years in prison if convicted on all the&nbsp;charges.</p><p>Chang Je-won, spokesperson for the Liberty Korea Party, the conservative opposition to Moon&#8217;s government, was on hand at Lee&#8217;s residence to watch the ex-president being taken away. And he released a statement expressing his party&#8217;s regret at the&nbsp;development.</p><p>&#8220;Should living as a former president in this country be this difficult?&#8221; he said. &#8220;I hope former President Lee will be the last victim of a political&nbsp;revenge.&#8221;</p><p>Prosecutors have 20 days to hold Lee at the Seoul Eastern Detention Center, while they decide whether to slap him with formal&nbsp;charges.</p><p><em>Producer Se Eun Gong contributed to this report.</em>  <div class="fullattribution">Copyright 2018 NPR.
